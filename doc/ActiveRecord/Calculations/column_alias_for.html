ActiveRecord::Calculations#column_alias_for

Converts the given keys to the value that the database adapter returns as a usable column name: column_alias_for("users.id") # => "users_id" column_alias_for("sum(id)") # => "sum_id" column_alias_for("count(distinct users.id)") # => "count_distinct_users_id" column_alias_for("count(*)") # => "count_all" column_alias_for("count", "id") # => "count_id"
