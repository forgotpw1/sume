ActionView::Helpers::DateHelper#time_ago_in_words

Like distance_of_time_in_words, but where to_time is fixed to Time.now. time_ago_in_words(3.minutes.from_now) # => 3 minutes time_ago_in_words(Time.now - 15.hours) # => about 15 hours time_ago_in_words(Time.now) # => less than a minute time_ago_in_words(Time.now, :include_seconds => true) # => less than 5 seconds from_time = Time.now - 3.days - 14.minutes - 25.seconds time_ago_in_words(from_time) # => 3 days
